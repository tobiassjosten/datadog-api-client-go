    16  // IncidentResponseAttributes The incident's attributes from a response.
    17  type IncidentResponseAttributes struct {
    18  	// Timestamp when the incident was created.
    19  	Created *time.Time `json:"created,omitempty"`
    20  	// Length of the incident's customer impact in seconds.
    21  	// Equals the difference between `customer_impact_start` and `customer_impact_end`.
    22  	CustomerImpactDuration *int64 `json:"customer_impact_duration,omitempty"`
    23  	// Timestamp when customers were no longer impacted by the incident.
    24  	CustomerImpactEnd datadog.NullableTime `json:"customer_impact_end,omitempty"`
    25  	// A summary of the impact customers experienced during the incident.
    26  	CustomerImpactScope datadog.NullableString `json:"customer_impact_scope,omitempty"`
    27  	// Timestamp when customers began being impacted by the incident.
    28  	CustomerImpactStart datadog.NullableTime `json:"customer_impact_start,omitempty"`
    29  	// A flag indicating whether the incident caused customer impact.
    30  	CustomerImpacted *bool `json:"customer_impacted,omitempty"`
    31  	// Timestamp when the incident was detected.
    32  	Detected datadog.NullableTime `json:"detected,omitempty"`
    33  	// A condensed view of the user-defined fields attached to incidents.
    34  	Fields map[string]IncidentFieldAttributes `json:"fields,omitempty"`
    35  	// Timestamp when the incident was last modified.
    36  	Modified *time.Time `json:"modified,omitempty"`
    37  	// Notification handles that will be notified of the incident during update.
    38  	NotificationHandles []IncidentNotificationHandle `json:"notification_handles,omitempty"`
    39  	// The monotonically increasing integer ID for the incident.
    40  	PublicId *int64 `json:"public_id,omitempty"`
    41  	// Timestamp when the incident's state was last changed from active or stable to resolved or completed.
    42  	Resolved datadog.NullableTime `json:"resolved,omitempty"`
    43  	// The amount of time in seconds to detect the incident.
    44  	// Equals the difference between `customer_impact_start` and `detected`.
    45  	TimeToDetect *int64 `json:"time_to_detect,omitempty"`
    46  	// The amount of time in seconds to call incident after detection. Equals the difference of `detected` and `created`.
    47  	TimeToInternalResponse *int64 `json:"time_to_internal_response,omitempty"`
    48  	// The amount of time in seconds to resolve customer impact after detecting the issue. Equals the difference between `customer_impact_end` and `detected`.
    49  	TimeToRepair *int64 `json:"time_to_repair,omitempty"`
    50  	// The amount of time in seconds to resolve the incident after it was created. Equals the difference between `created` and `resolved`.
    51  	TimeToResolve *int64 `json:"time_to_resolve,omitempty"`
    52  	// The title of the incident, which summarizes what happened.
    53  	Title string `json:"title"`
    54  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    55  	UnparsedObject       map[string]interface{} `json:"-"`
    56  	AdditionalProperties map[string]interface{}
    57  }
