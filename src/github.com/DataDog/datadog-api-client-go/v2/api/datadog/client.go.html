    89  // ParameterToString convert interface{} parameters to string, using a delimiter if format is provided.
    90  func ParameterToString(obj interface{}, collectionFormat string) string {
    91  	var delimiter string
    92  
    93  	switch collectionFormat {
    94  	case "pipes":
    95  		delimiter = "|"
    96  	case "ssv":
    97  		delimiter = " "
    98  	case "tsv":
    99  		delimiter = "\t"
   100  	case "csv":
   101  		delimiter = ","
   102  	}
   103  
   104  	if reflect.TypeOf(obj).Kind() == reflect.Slice {
   105  		return strings.Trim(strings.Replace(fmt.Sprint(obj), " ", delimiter, -1), "[]")
   106  	} else if t, ok := obj.(time.Time); ok {
   107  		if t.Nanosecond() == 0 {
   108  			return t.Format("2006-01-02T15:04:05Z07:00")
   109  		}
   110  		return t.Format("2006-01-02T15:04:05.000Z07:00")
   111  	}
   112  
   113  	return fmt.Sprintf("%v", obj)
   114  }

   116  // CallAPI do the request.
   117  func (c *APIClient) CallAPI(request *http.Request) (*http.Response, error) {
   118  	if c.Cfg.Debug {
   119  		dump, err := httputil.DumpRequestOut(request, true)
   120  		if err != nil {
   121  			return nil, err
   122  		}
   123  		// Strip any api keys from the response being logged
   124  		keys, ok := request.Context().Value(ContextAPIKeys).(map[string]APIKey)
   125  		if keys != nil && ok {
   126  			for _, apiKey := range keys {
   127  				valueRegex := regexp.MustCompile(fmt.Sprintf("(?m)%s", apiKey.Key))
   128  				dump = valueRegex.ReplaceAll(dump, []byte("REDACTED"))
   129  			}
   130  		}
   131  		log.Printf("\n%s\n", string(dump))
   132  	}
   133  
   134  	resp, err := c.Cfg.HTTPClient.Do(request)
   135  	if err != nil {
   136  		return resp, err
   137  	}
   138  
   139  	if c.Cfg.Debug {
   140  		dump, err := httputil.DumpResponse(resp, true)
   141  		if err != nil {
   142  			return resp, err
   143  		}
   144  		log.Printf("\n%s\n", string(dump))
   145  	}
   146  	return resp, err
   147  }

   155  // PrepareRequest build the request.
   156  func (c *APIClient) PrepareRequest(
   157  	ctx context.Context,
   158  	path string, method string,
   159  	postBody interface{},
   160  	headerParams map[string]string,
   161  	queryParams url.Values,
   162  	formParams url.Values,
   163  	formFile *FormFile) (localVarRequest *http.Request, err error) {
   164  
   165  	var body *bytes.Buffer
   166  
   167  	// Detect postBody type and post.
   168  	if postBody != nil {
   169  		contentType := headerParams["Content-Type"]
   170  		if contentType == "" {
   171  			contentType = detectContentType(postBody)
   172  			headerParams["Content-Type"] = contentType
   173  		}
   174  
   175  		body, err = setBody(postBody, contentType)
   176  		if err != nil {
   177  			return nil, err
   178  		}
   179  	}
   180  
   181  	// add form parameters and file if available.
   182  	if strings.HasPrefix(headerParams["Content-Type"], "multipart/form-data") && len(formParams) > 0 || formFile != nil {
   183  		if body != nil {
   184  			return nil, errors.New("cannot specify postBody and multipart form at the same time")
   185  		}
   186  		body = &bytes.Buffer{}
   187  		w := multipart.NewWriter(body)
   188  
   189  		for k, v := range formParams {
   190  			for _, iv := range v {
   191  				if strings.HasPrefix(k, "@") { // file
   192  					err = addFile(w, k[1:], iv)
   193  					if err != nil {
   194  						return nil, err
   195  					}
   196  				} else { // form value
   197  					w.WriteField(k, iv)
   198  				}
   199  			}
   200  		}
   201  		if formFile != nil {
   202  			w.Boundary()
   203  			part, err := w.CreateFormFile(formFile.FormFileName, filepath.Base(formFile.FileName))
   204  			if err != nil {
   205  				return nil, err
   206  			}
   207  			_, err = part.Write(formFile.FileBytes)
   208  			if err != nil {
   209  				return nil, err
   210  			}
   211  		}
   212  
   213  		// Set the Boundary in the Content-Type
   214  		headerParams["Content-Type"] = w.FormDataContentType()
   215  
   216  		// Set Content-Length
   217  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   218  		w.Close()
   219  	}
   220  
   221  	if strings.HasPrefix(headerParams["Content-Type"], "application/x-www-form-urlencoded") && len(formParams) > 0 {
   222  		if body != nil {
   223  			return nil, errors.New("cannot specify postBody and x-www-form-urlencoded form at the same time")
   224  		}
   225  		body = &bytes.Buffer{}
   226  		body.WriteString(formParams.Encode())
   227  		// Set Content-Length
   228  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   229  	}
   230  
   231  	// Setup path and query parameters
   232  	url, err := url.Parse(path)
   233  	if err != nil {
   234  		return nil, err
   235  	}
   236  
   237  	// Override request host, if applicable
   238  	if c.Cfg.Host != "" {
   239  		url.Host = c.Cfg.Host
   240  	}
   241  
   242  	// Override request scheme, if applicable
   243  	if c.Cfg.Scheme != "" {
   244  		url.Scheme = c.Cfg.Scheme
   245  	}
   246  
   247  	// Adding Query Param
   248  	query := url.Query()
   249  	for k, v := range queryParams {
   250  		for _, iv := range v {
   251  			query.Add(k, iv)
   252  		}
   253  	}
   254  
   255  	// Encode the parameters.
   256  	url.RawQuery = query.Encode()
   257  
   258  	// Generate a new request
   259  	if body != nil {
   260  		if headerParams["Content-Encoding"] == "gzip" {
   261  			var buf bytes.Buffer
   262  			compressor := gzip.NewWriter(&buf)
   263  			if _, err = compressor.Write(body.Bytes()); err != nil {
   264  				return nil, err
   265  			}
   266  			if err = compressor.Close(); err != nil {
   267  				return nil, err
   268  			}
   269  			body = &buf
   270  
   271  		} else if headerParams["Content-Encoding"] == "deflate" {
   272  			var buf bytes.Buffer
   273  			compressor := zlib.NewWriter(&buf)
   274  			if _, err = compressor.Write(body.Bytes()); err != nil {
   275  				return nil, err
   276  			}
   277  			if err = compressor.Close(); err != nil {
   278  				return nil, err
   279  			}
   280  			body = &buf
   281  		} else if headerParams["Content-Encoding"] == "zstd1" {
   282  			body, err = compressZstd(body.Bytes())
   283  			if err != nil {
   284  				return nil, err
   285  			}
   286  		}
   287  		headerParams["Content-Length"] = fmt.Sprintf("%d", body.Len())
   288  		localVarRequest, err = http.NewRequest(method, url.String(), body)
   289  	} else {
   290  		localVarRequest, err = http.NewRequest(method, url.String(), nil)
   291  	}
   292  	if err != nil {
   293  		return nil, err
   294  	}
   295  
   296  	// add header parameters, if any
   297  	if len(headerParams) > 0 {
   298  		headers := http.Header{}
   299  		for h, v := range headerParams {
   300  			headers.Set(h, v)
   301  		}
   302  		localVarRequest.Header = headers
   303  	}
   304  
   305  	// Add the user agent to the request.
   306  	localVarRequest.Header.Add("User-Agent", c.Cfg.UserAgent)
   307  
   308  	if ctx != nil {
   309  		// add context to the request
   310  		localVarRequest = localVarRequest.WithContext(ctx)
   311  
   312  		// Walk through any authentication.
   313  
   314  		// OAuth2 authentication
   315  		if tok, ok := ctx.Value(ContextOAuth2).(oauth2.TokenSource); ok {
   316  			// We were able to grab an oauth2 token from the context
   317  			var latestToken *oauth2.Token
   318  			if latestToken, err = tok.Token(); err != nil {
   319  				return nil, err
   320  			}
   321  
   322  			latestToken.SetAuthHeader(localVarRequest)
   323  		}
   324  
   325  		// Basic HTTP Authentication
   326  		if auth, ok := ctx.Value(ContextBasicAuth).(BasicAuth); ok {
   327  			localVarRequest.SetBasicAuth(auth.UserName, auth.Password)
   328  		}
   329  
   330  		// AccessToken Authentication
   331  		if auth, ok := ctx.Value(ContextAccessToken).(string); ok {
   332  			localVarRequest.Header.Add("Authorization", "Bearer "+auth)
   333  		}
   334  	}
   335  
   336  	for header, value := range c.Cfg.DefaultHeader {
   337  		localVarRequest.Header.Add(header, value)
   338  	}
   339  
   340  	if !c.Cfg.Compress {
   341  		// gzip is on by default, so disable it by setting encoding to identity
   342  		localVarRequest.Header.Add("Accept-Encoding", "identity")
   343  	}
   344  	return localVarRequest, nil
   345  }

   347  // Decode unmarshal bytes into an interface
   348  func (c *APIClient) Decode(v interface{}, b []byte, contentType string) (err error) {
   349  	if len(b) == 0 {
   350  		return nil
   351  	}
   352  	if s, ok := v.(*string); ok {
   353  		*s = string(b)
   354  		return nil
   355  	}
   356  	if xmlCheck.MatchString(contentType) {
   357  		if err = xml.Unmarshal(b, v); err != nil {
   358  			return err
   359  		}
   360  		return nil
   361  	}
   362  	if actualObj, ok := v.(interface{ GetActualInstance() interface{} }); ok { // oneOf, anyOf schemas
   363  		if unmarshalObj, ok := actualObj.(interface{ UnmarshalJSON([]byte) error }); ok { // make sure it has UnmarshalJSON defined
   364  			if err = unmarshalObj.UnmarshalJSON(b); err != nil {
   365  				return err
   366  			}
   367  		} else {
   368  			return errors.New("unknown type with GetActualInstance but no unmarshalObj.UnmarshalJSON defined")
   369  		}
   370  	} else if err = json.Unmarshal(b, v); err != nil { // simple model
   371  		return err
   372  	}
   373  	return nil
   374  }

   386  		return err
   387  	}
   388  	_, err = io.Copy(part, file)
   389  
   390  	return err

   398  // Set request body from an interface{}.
   399  func setBody(body interface{}, contentType string) (bodyBuf *bytes.Buffer, err error) {
   400  	if reflect.ValueOf(body).IsNil() {
   401  		return nil, nil
   402  	}
   403  
   404  	if bodyBuf == nil {
   405  		bodyBuf = &bytes.Buffer{}
   406  	}
   407  
   408  	if reader, ok := body.(io.Reader); ok {
   409  		_, err = bodyBuf.ReadFrom(reader)
   410  	} else if fp, ok := body.(**os.File); ok {
   411  		_, err = bodyBuf.ReadFrom(*fp)
   412  	} else if b, ok := body.([]byte); ok {
   413  		_, err = bodyBuf.Write(b)
   414  	} else if s, ok := body.(string); ok {
   415  		_, err = bodyBuf.WriteString(s)
   416  	} else if s, ok := body.(*string); ok {
   417  		_, err = bodyBuf.WriteString(*s)
   418  	} else if jsonCheck.MatchString(contentType) {
   419  		err = json.NewEncoder(bodyBuf).Encode(body)
   420  	} else if xmlCheck.MatchString(contentType) {
   421  		err = xml.NewEncoder(bodyBuf).Encode(body)
   422  	}
   423  
   424  	if err != nil {
   425  		return nil, err
   426  	}
   427  
   428  	if bodyBuf.Len() == 0 {
   429  		return nil, fmt.Errorf("invalid body type %s", contentType)
   430  	}
   431  	return bodyBuf, nil
   432  }

   434  // detectContentType method is used to figure out `Request.Body` content type for request header.
   435  func detectContentType(body interface{}) string {
   436  	contentType := "text/plain; charset=utf-8"
   437  	kind := reflect.TypeOf(body).Kind()
   438  
   439  	switch kind {
   440  	case reflect.Struct, reflect.Map, reflect.Ptr:
   441  		contentType = "application/json; charset=utf-8"
   442  	case reflect.String:
   443  		contentType = "text/plain; charset=utf-8"
   444  	default:
   445  		if b, ok := body.([]byte); ok {
   446  			contentType = http.DetectContentType(b)
   447  		} else if kind == reflect.Slice {
   448  			contentType = "application/json; charset=utf-8"
   449  		}
   450  	}
   451  
   452  	return contentType
   453  }
