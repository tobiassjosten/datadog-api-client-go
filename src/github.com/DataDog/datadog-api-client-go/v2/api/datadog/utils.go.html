   397  // ContainsUnparsedObject returns true if the given data contains an unparsed object from the API.
   398  func ContainsUnparsedObject(i interface{}) (bool, interface{}) {
   399  	v := reflect.ValueOf(i)
   400  	switch v.Kind() {
   401  	case reflect.Array, reflect.Slice:
   402  		for i := 0; i < v.Len(); i++ {
   403  			if n, m := ContainsUnparsedObject(v.Index(i).Interface()); n {
   404  				return n, m
   405  			}
   406  		}
   407  	case reflect.Map:
   408  		for _, k := range v.MapKeys() {
   409  			if n, m := ContainsUnparsedObject(v.MapIndex(k).Interface()); n {
   410  				return n, m
   411  			}
   412  		}
   413  	case reflect.Struct:
   414  		if u := v.FieldByName("UnparsedObject"); u.IsValid() && !u.IsNil() {
   415  			return true, u.Interface()
   416  		}
   417  		for i := 0; i < v.NumField(); i++ {
   418  			if fn := v.Type().Field(i).Name; string(fn[0]) == strings.ToUpper(string(fn[0])) && fn != "UnparsedObject" {
   419  				if n, m := ContainsUnparsedObject(v.Field(i).Interface()); n {
   420  					return n, m
   421  				}
   422  			} else if fn == "value" { // Special case for Nullables
   423  				if get := v.MethodByName("Get"); get.IsValid() {
   424  					if n, m := ContainsUnparsedObject(get.Call([]reflect.Value{})[0].Interface()); n {
   425  						return n, m
   426  					}
   427  				}
   428  			}
   429  		}
   430  	case reflect.Interface, reflect.Ptr:
   431  		if !v.IsNil() {
   432  			return ContainsUnparsedObject(v.Elem().Interface())
   433  		}
   434  	default:
   435  		if v.IsValid() {
   436  			if m := v.MethodByName("IsValid"); m.IsValid() {
   437  				if !m.Call([]reflect.Value{})[0].Bool() {
   438  					return true, v.Interface()
   439  				}
   440  			}
   441  		}
   442  	}
   443  	return false, nil
   444  }
