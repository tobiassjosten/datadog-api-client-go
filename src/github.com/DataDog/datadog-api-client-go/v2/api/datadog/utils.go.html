   396  // ContainsUnparsedObject returns true if the given data contains an unparsed object from the API.
   397  func ContainsUnparsedObject(i interface{}) (bool, interface{}) {
   398  	v := reflect.ValueOf(i)
   399  	switch v.Kind() {
   400  	case reflect.Array, reflect.Slice:
   401  		for i := 0; i < v.Len(); i++ {
   402  			if n, m := ContainsUnparsedObject(v.Index(i).Interface()); n {
   403  				return n, m
   404  			}
   405  		}
   406  	case reflect.Map:
   407  		for _, k := range v.MapKeys() {
   408  			if n, m := ContainsUnparsedObject(v.MapIndex(k).Interface()); n {
   409  				return n, m
   410  			}
   411  		}
   412  	case reflect.Struct:
   413  		if u := v.FieldByName("UnparsedObject"); u.IsValid() && !u.IsNil() {
   414  			return true, u.Interface()
   415  		}
   416  		for i := 0; i < v.NumField(); i++ {
   417  			if fn := v.Type().Field(i).Name; string(fn[0]) == strings.ToUpper(string(fn[0])) && fn != "UnparsedObject" {
   418  				if n, m := ContainsUnparsedObject(v.Field(i).Interface()); n {
   419  					return n, m
   420  				}
   421  			} else if fn == "value" { // Special case for Nullables
   422  				if get := v.MethodByName("Get"); get.IsValid() {
   423  					if n, m := ContainsUnparsedObject(get.Call([]reflect.Value{})[0].Interface()); n {
   424  						return n, m
   425  					}
   426  				}
   427  			}
   428  		}
   429  	case reflect.Interface, reflect.Ptr:
   430  		if !v.IsNil() {
   431  			return ContainsUnparsedObject(v.Elem().Interface())
   432  		}
   433  	default:
   434  		if v.IsValid() {
   435  			if m := v.MethodByName("IsValid"); m.IsValid() {
   436  				if !m.Call([]reflect.Value{})[0].Bool() {
   437  					return true, v.Interface()
   438  				}
   439  			}
   440  		}
   441  	}
   442  	return false, nil
   443  }
