   452  // ContainsUnparsedObject returns true if the given data contains an unparsed object from the API.
   453  func ContainsUnparsedObject(i interface{}) (bool, interface{}) {
   454  	v := reflect.ValueOf(i)
   455  	switch v.Kind() {
   456  	case reflect.Array, reflect.Slice:
   457  		for i := 0; i < v.Len(); i++ {
   458  			if n, m := ContainsUnparsedObject(v.Index(i).Interface()); n {
   459  				return n, m
   460  			}
   461  		}
   462  	case reflect.Map:
   463  		for _, k := range v.MapKeys() {
   464  			if n, m := ContainsUnparsedObject(v.MapIndex(k).Interface()); n {
   465  				return n, m
   466  			}
   467  		}
   468  	case reflect.Struct:
   469  		if u := v.FieldByName("UnparsedObject"); u.IsValid() && !u.IsNil() {
   470  			return true, u.Interface()
   471  		}
   472  		for i := 0; i < v.NumField(); i++ {
   473  			if fn := v.Type().Field(i).Name; string(fn[0]) == strings.ToUpper(string(fn[0])) && fn != "UnparsedObject" {
   474  				if n, m := ContainsUnparsedObject(v.Field(i).Interface()); n {
   475  					return n, m
   476  				}
   477  			} else if fn == "value" { // Special case for Nullables
   478  				if get := v.MethodByName("Get"); get.IsValid() {
   479  					if n, m := ContainsUnparsedObject(get.Call([]reflect.Value{})[0].Interface()); n {
   480  						return n, m
   481  					}
   482  				}
   483  			}
   484  		}
   485  	case reflect.Interface, reflect.Ptr:
   486  		if !v.IsNil() {
   487  			return ContainsUnparsedObject(v.Elem().Interface())
   488  		}
   489  	default:
   490  		if v.IsValid() {
   491  			if m := v.MethodByName("IsValid"); m.IsValid() {
   492  				if !m.Call([]reflect.Value{})[0].Bool() {
   493  					return true, v.Interface()
   494  				}
   495  			}
   496  		}
   497  	}
   498  	return false, nil
   499  }

   506  // Copy the original request so it doesn't get lost when retrying
   507  func copyRequest(r *http.Request, rawBody *[]byte) *http.Request {
   508  	newRequest := *r
   509  
   510  	if r.Body == nil || r.Body == http.NoBody {
   511  		return &newRequest
   512  	}
   513  	newRequest.Body = io.NopCloser(bytes.NewBuffer(*rawBody))
   514  	return &newRequest
   515  }
