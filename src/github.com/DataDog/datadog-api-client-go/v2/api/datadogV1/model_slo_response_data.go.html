    13  // SLOResponseData A service level objective object includes a service level indicator, thresholds
    14  // for one or more timeframes, and metadata (`name`, `description`, `tags`, etc.).
    15  type SLOResponseData struct {
    16  	// A list of SLO monitors IDs that reference this SLO. This field is returned only when `with_configured_alert_ids` parameter is true in query.
    17  	ConfiguredAlertIds []int64 `json:"configured_alert_ids,omitempty"`
    18  	// Creation timestamp (UNIX time in seconds)
    19  	//
    20  	// Always included in service level objective responses.
    21  	CreatedAt *int64 `json:"created_at,omitempty"`
    22  	// Object describing the creator of the shared element.
    23  	Creator *Creator `json:"creator,omitempty"`
    24  	// A user-defined description of the service level objective.
    25  	//
    26  	// Always included in service level objective responses (but may be `null`).
    27  	// Optional in create/update requests.
    28  	Description datadog.NullableString `json:"description,omitempty"`
    29  	// A list of (up to 20) monitor groups that narrow the scope of a monitor service level objective.
    30  	//
    31  	// Included in service level objective responses if it is not empty. Optional in
    32  	// create/update requests for monitor service level objectives, but may only be
    33  	// used when then length of the `monitor_ids` field is one.
    34  	Groups []string `json:"groups,omitempty"`
    35  	// A unique identifier for the service level objective object.
    36  	//
    37  	// Always included in service level objective responses.
    38  	Id *string `json:"id,omitempty"`
    39  	// Modification timestamp (UNIX time in seconds)
    40  	//
    41  	// Always included in service level objective responses.
    42  	ModifiedAt *int64 `json:"modified_at,omitempty"`
    43  	// A list of monitor ids that defines the scope of a monitor service level
    44  	// objective. **Required if type is `monitor`**.
    45  	MonitorIds []int64 `json:"monitor_ids,omitempty"`
    46  	// The union of monitor tags for all monitors referenced by the `monitor_ids`
    47  	// field.
    48  	// Always included in service level objective responses for monitor service level
    49  	// objectives (but may be empty). Ignored in create/update requests. Does not
    50  	// affect which monitors are included in the service level objective (that is
    51  	// determined entirely by the `monitor_ids` field).
    52  	MonitorTags []string `json:"monitor_tags,omitempty"`
    53  	// The name of the service level objective object.
    54  	Name *string `json:"name,omitempty"`
    55  	// A metric-based SLO. **Required if type is `metric`**. Note that Datadog only allows the sum by aggregator
    56  	// to be used because this will sum up all request counts instead of averaging them, or taking the max or
    57  	// min of all of those requests.
    58  	Query *ServiceLevelObjectiveQuery `json:"query,omitempty"`
    59  	// A list of tags associated with this service level objective.
    60  	// Always included in service level objective responses (but may be empty).
    61  	// Optional in create/update requests.
    62  	Tags []string `json:"tags,omitempty"`
    63  	// The target threshold such that when the service level indicator is above this
    64  	// threshold over the given timeframe, the objective is being met.
    65  	TargetThreshold *float64 `json:"target_threshold,omitempty"`
    66  	// The thresholds (timeframes and associated targets) for this service level
    67  	// objective object.
    68  	Thresholds []SLOThreshold `json:"thresholds,omitempty"`
    69  	// The SLO time window options.
    70  	Timeframe *SLOTimeframe `json:"timeframe,omitempty"`
    71  	// The type of the service level objective.
    72  	Type *SLOType `json:"type,omitempty"`
    73  	// The optional warning threshold such that when the service level indicator is
    74  	// below this value for the given threshold, but above the target threshold, the
    75  	// objective appears in a "warning" state. This value must be greater than the target
    76  	// threshold.
    77  	WarningThreshold *float64 `json:"warning_threshold,omitempty"`
    78  	// UnparsedObject contains the raw value of the object if there was an error when deserializing into the struct
    79  	UnparsedObject       map[string]interface{} `json:"-"`
    80  	AdditionalProperties map[string]interface{}
    81  }
